@FormatIf HTML
<a name="Break"></a>
@FormatEndIf
@Syntax

  @Keyword Break [Niveau]

@Description
  @Keyword Break permet de quitter à n'importe 
  quel moment une ou plusieurs des boucles 
  suivantes: 
  @ReferenceLink "Repeat_Until" Repeat, 
  @ReferenceLink "For_Next" For, 
  @ReferenceLink "ForEach_Next" ForEach 
  et @ReferenceLink "While_Wend" While.

@OptionalParameter "Niveau"  
  Indique le nombre de boucles qui doivent être 
  abrégées. @Linebreak
  Sans paramètre, @Keyword Break quitte 
  la boucle en cours.

@NoReturnValue

@Example Boucle simple
@Code        
  For k=0 To 10
    If k=5
      Break  ; Quitte immédiatement la boucle For/Next
    EndIf
    Debug k
  Next
@EndCode   

  @Example Boucles imbriquées
  
@Code   
  For k=0 To 10
    Counter = 0
    Repeat
      If k=5
        Break 2 ; Quitte immédiatement les boucles Repeat/Until et For/Next
      EndIf
      Counter+1
    Until Counter > 1
    Debug k 
  Next
@EndCode

<a name="Continue"></a>
@FormatEndIf
@Syntax

  @Keyword Continue

@Description
  @Keyword Continue permet de passer directement 
  à la prochaine itération dans l'une des boucles 
  suivantes: 
  @ReferenceLink "Repeat_Until" Repeat, 
  @ReferenceLink "For_Next" For, 
  @ReferenceLink "ForEach_Next" ForEach et 
  @ReferenceLink "While_Wend" While. 

@NoParameters

@NoReturnValue
  
@Example
@Code       
  For k=0 To 10
    If k=5
      Continue  ; N'affichera pas 'Debug 5' car la fin de cette itération sera ignorée
    EndIf
    Debug k
  Next
@EndCode
